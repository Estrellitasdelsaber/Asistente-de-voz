recognition.onresult = (e) => { let texto = e.results[0][0].transcript.toLowerCase(); salida.textContent = 'Escuché: ' + texto; // Saca solo los números let nums = texto.match(/\d+/g); if(nums && nums.length >= 2){ let n1 = parseInt(nums[0]); let n2 = parseInt(nums[1]); if(texto.includes('más') || texto.includes('suma')){ salida.textContent = n1 + ' + ' + n2 + ' = ' + (n1+n2); } if(texto.includes('menos') || texto.includes('resta')){ salida.textContent = n1 + ' - ' + n2 + ' = ' + (n1-n2); } if(texto.includes('por') || texto.includes('multiplica')){ salida.textContent = n1 + ' x ' + n2 + ' = ' + (n1*n2); } } };
